def rec_structure(menu_node)
  @@rec_dep+=1
  h = menu_node.get_hash_node
  if @@rec_dep > 1
    p @@rec_dep
    sum = '<br>'
    @@rec_dep.times{
      sum += '-'
    }
    @@rez_out +=  '|' + sum + "#{h[:name]}"
    #p  '|' + sum + "#{h[:url]}"
  else
    @@rez_out +=  "|#{h[:name]}"
    #p "|#{h[:url]}"
  end

  menu_node.get_children.each{|child|
    rec_structure(child)
  }
end
